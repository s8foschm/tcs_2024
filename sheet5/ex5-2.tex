\section{Syntactical sugar}

\subsection{(a)}

Statement: \( x_i := x_j^{x_k} \)

\begin{algorithmic}[1]
\State \( x_i := 1; \)
\For{\( x_k \)}
    \State \( x := 0; \)
    \For{\( x_j \)}
        \State \( x := x + x_i; \)
    \EndFor \textbf{od;}
    \State \( x_i := x; \)
\EndFor \textbf{od;}
\end{algorithmic}


\subsection{(b)}

\begin{algorithmic}[1]
\State \( x_{cmp} := 1; \)  
\For{\( x_i \)}
    \State \( x_{cmp} := 0; \) 
\EndFor \textbf{od;}
\For{\( x_{cmp} \)}
     \( P_1; \)
\EndFor \textbf{od;}
\For{\( x_i \)}
    \State\( P_2; \)
\EndFor \textbf{od;}
\end{algorithmic}


\subsection{(c)}
We can simulate For FOR-programs with WHILE-programs:

\begin{algorithmic}[1]
\State \( x_i := 0; \)
\While{\( x_j \geq x_k \)}
    \State \( x_j := x_j - x_k; \)
    \State \( x_i++; \)
\EndWhile \textbf{od;}
\end{algorithmic}
\subsection{(d)}
Here as well, we can simulate FOR-programs with WHILE-programs:

\begin{algorithmic}[1]
\While{\( x_j \geq x_k \)}
    \State \( x_j := x_j - x_k; \)
\EndWhile \textbf{od;}
\State \( x_i := x_j; \)
\end{algorithmic}

\subsection{(e)}

\begin{algorithmic}[1]
\State \( x_k := x_i \circ x_j; \) \texttt{/* assuming \( x_i \circ x_j \) : 0 if true and 1 if false */}
\If{\( x_k = 0 \)} 
    \State \( x := 0; \)
\Else 
    \State \( x := 1; \)
\EndIf \textbf{fi;}
\For{\( x \)} 
    \State  \( P_1; \)
\EndFor \textbf{od;}
\For{\( 1 - x \)}
    \State  \( P_2; \)
\EndFor \textbf{od;}
\end{algorithmic}
