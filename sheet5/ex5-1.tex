\section*{FOR-Programs}

\subsection*{(a)}
\subsubsection*{Statement: \( x_i := x_j + x_k \)}
\label{alg:addition}

\begin{algorithmic}[1]
\State \( x_i := 0; \)
\For{\( x_j \)}
    \State \( x_i \texttt{++}; \)
\EndFor \textbf{od;}
\For{\( x_k \)}
    \State \( x_i \texttt{++}; \)
\EndFor \textbf{od;}
\end{algorithmic}

\subsubsection*{Statement: \( x_i := x_j - x_k \)}
\label{alg:subtraction}

\begin{algorithmic}[1]
\State \( x_i := 0; \)
\For{\( x_j \)}
    \State \( x_i \texttt{++}; \)
\EndFor \textbf{od;}
\For{\( x_k \)}
    \State \( x_i \texttt{--}; \)
\EndFor \textbf{od;}
\end{algorithmic}

\subsubsection*{Statement: \( x_i := c \)}
\label{alg:equality}

\begin{algorithmic}[1]
\State \( x_i := 0; \)
\State \( x_i \texttt{++}; \)
\State \( x_i \texttt{++}; \)
\State \dots
\State \( x_i \texttt{++}; \) \texttt{/* the \( c \)-th \( x_i \texttt{++} \) statement */}
\end{algorithmic}


\subsection*{(b)}
\subsubsection*{Statement: \( x_i \texttt{--} \)}

\label{alg:decrement}

\begin{algorithmic}[1]
\State \( x_k := 0; \)
\For{\( x_i \)}
    \State \( x_i := 0; \)
    \For{\( x_k \)}
        \State \( x_i \texttt{++}; \)
    \EndFor \textbf{od;}
    \State \( x_k \texttt{++}; \)
\EndFor \textbf{od;}
\end{algorithmic}
